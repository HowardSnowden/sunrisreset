def create
  @record = EmailProp.new(email_params)
  respond_to do |format|
   if @record.save
    
    format.js { render :action=> 'create'}
    ModelMailer.incoming_messages(@record).deliver

   else 
     
       format.json { render :json => { :error => @record.errors.full_messages }, :status => 422 }
            
   end
  end
end
